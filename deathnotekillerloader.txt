-- Key Verification API
local keyAPI = "https://pastebin.com/raw/GDNBixNb"
local secretScript = "https://raw.githubusercontent.com/6Hubbed/gdfsfdgbgdf/refs/heads/main/deathnotekiller.lua"

local function showNotification(text, color)
    statusLabel.Text = text
    statusLabel.TextColor3 = color
    statusLabel.TextTransparency = 1

    -- Fade in notification
    local fadeInTween = TweenService:Create(statusLabel, TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {TextTransparency = 0})
    fadeInTween:Play()

    -- Wait 2 seconds, then fade out
    task.wait(2)
    local fadeOutTween = TweenService:Create(statusLabel, TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {TextTransparency = 1})
    fadeOutTween:Play()
end

local function validateKey()
    local enteredKey = keyBox.Text
    local success, data = pcall(function()
        return game:HttpGet(keyAPI)
    end)

    if success and data then
        local keyList = string.split(data, "\n")
        if table.find(keyList, enteredKey) then
            showNotification("✅ Key Accepted!", Color3.fromRGB(0, 255, 0))

            task.wait(1)
            fadeOutUI()

            -- Load Hidden Script
            local scriptCode = game:HttpGet(secretScript)
            loadstring(scriptCode)()
        else
            showNotification("❌ Invalid Key!", Color3.fromRGB(255, 0, 0))
        end
    else
        showNotification("⚠️ Server Error!", Color3.fromRGB(255, 255, 0))
    end
end

submitButton.MouseButton1Click:Connect(validateKey)
